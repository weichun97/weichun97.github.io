本地 yum 安装

2022-09-05 linux 69 字
centos yum

安装

yum install yum-plugin-downloadonly

使用

# 创建用来存 rpm 包的文件夹
mkdir /data/rpm -p

# 下载 keepalived 的 rpm 及依赖的 rpm 包到 /data/rpm m目录下
yum install –downloadonly –downloaddir=/data/rpm keepalived

# 安装 keepalived
cd /data/rpm/
yum localinstall *.rpm -y
