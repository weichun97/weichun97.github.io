nginx高可用搭建

2022-09-05 linux 136 字
nginx keepalived

环境信息

ip 软件
192.168.1.201 nginx、keepalived
192.168.1.202 nginx、keepalived

keepalived 配置

master

主节点配置

global_defs {
  # Keepalived process identifier
  router_id nginx
}

# Script to check whether Nginx is running or not
vrrp_script check_nginx {
  script "/bin/check_nginx.sh"
  interval 2
  weight 50
}

# state: MASTER(主)/BACKUP(备)
# interface: 网卡，ip add 命令查看
# virtual_router_id: vrrp_instance的唯一ID
vrrp_instance VI_01 {
  state MASTER
  interface enp0s3
  virtual_router_id 151
  priority 110

  # 虚拟ip
  virtual_ipaddress {
    192.168.31.50/24
  }
  # 调用的脚本
  track_script {
    check_nginx
  }
  authentication {
    auth_type AH
    auth_pass secret
  }
}
